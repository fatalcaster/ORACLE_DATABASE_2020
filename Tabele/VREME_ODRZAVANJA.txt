CREATE TABLE VREME_ODRZAVANJA
(
    ID NUMBER(9) PRIMARY KEY,
    DATUM DATE DEFAULT SYSDATE NOT NULL,
    VREME INTERVAL DAY TO SECOND,
    ID_PREDSTAVE NUMBER(9),
    ID_KOMADA NUMBER(9),
    CONSTRAINT ID_PREDSTAVE_FK FOREIGN 
    KEY(ID_PREDSTAVE) REFERENCES PREDSTAVA(ID_PREDSTAVE),
    CONSTRAINT ID_KOMADA_FK FOREIGN 
    KEY(ID_KOMADA) REFERENCES POZORISNI_KOMAD(ID_KOMADA)
);
