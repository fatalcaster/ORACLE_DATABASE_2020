CREATE TABLE PREDSTAVA
(
    ID_PREDSTAVE NUMBER(9) CONSTRAINT ID_PREDSTAVE_PK PRIMARY KEY,
    DATUM_ODRZAVANJA DATE DEFAULT SYSDATE NOT NULL,
    BROJ_MESTA NUMBER(4),
    CENA_KARTE NUMBER(4),
    JMBG_PRODUCENTA NUMBER(13),
    ID_TRUPE NUMBER(9),
    CONSTRAINT FK_JMBG_PRODUCENTA FOREIGN 
    KEY(JMBG_PRODUCENTA) REFERENCES PRODUCENT(JMBG),
    CONSTRAINT FK_ID_TRUPE FOREIGN 
    KEY(ID_TRUPE) REFERENCES POZORISNA_TRUPA(ID_TRUPE)
);
