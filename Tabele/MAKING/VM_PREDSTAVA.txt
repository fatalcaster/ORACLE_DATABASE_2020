CREATE TABLE VM_PREDSTAVA
(
    ID NUMBER(9) PRIMARY KEY,
    DATUM_ODRZAVANJA DATE DEFAULT SYSDATE NOT NULL,
    BROJ_MESTA NUMBER(4),
    CENA_KARTE NUMBER(4),
    JMBG_PRODUCENTA NUMBER(13),
    ID_TRUPE NUMBER(9),
    CONSTRAINT PREDSTAVA_JMBG_PRODUCENTA FOREIGN 
    KEY(JMBG_PRODUCENTA) REFERENCES VM_PRODUCENT(JMBG),
    CONSTRAINT PREDSTAVA_ID_TRUPE FOREIGN 
    KEY(ID_TRUPE) REFERENCES VM_POZORISNA_TRUPA(ID)
);
