CREATE TABLE VM_UZRAST
(
    ID NUMBER(5) PRIMARY KEY,
    STAROSNA_GRANICA NUMBER(3),
    ID_PRETPLATNIKA NUMBER(9),
    ID_PREDSTAVE NUMBER(9),
    CONSTRAINT UZRAST_ID_PRETPLATNIKA FOREIGN 
    KEY(ID_PRETPLATNIKA) REFERENCES VM_PRETPLATNIK(ID),
    CONSTRAINT UZRAST_ID_PREDSTAVE_FK FOREIGN 
    KEY(ID_PREDSTAVE) REFERENCES VM_PREDSTAVA(ID)
);
