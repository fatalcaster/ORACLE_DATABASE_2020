CREATE TABLE VM_VREME_ODRZAVANJA
(
    ID NUMBER(9) PRIMARY KEY,
    DATUM DATE DEFAULT SYSDATE NOT NULL,
    VREME INTERVAL DAY TO SECOND,
    ID_PREDSTAVE NUMBER(9),
    ID_KOMADA NUMBER(9),
    CONSTRAINT VREME_ID_PREDSTAVE FOREIGN 
    KEY(ID_PREDSTAVE) REFERENCES VM_PREDSTAVA(ID),
    CONSTRAINT VREME_ID_KOMADA FOREIGN 
    KEY(ID_KOMADA) REFERENCES VM_POZORISNI_KOMAD(ID)
);
